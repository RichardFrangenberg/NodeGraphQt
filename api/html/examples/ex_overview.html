Simple Example¶

Here’s a basic example snippet for creating two nodes and connecting them together.

 1import sys
 2
 3from Qt import QtWidgets
 4from NodeGraphQt import NodeGraph, BaseNode, setup_context_menu
 5
 6
 7# create a node class object inherited from BaseNode.
 8class FooNode(BaseNode):
 9
10    # unique node identifier domain.
11    __identifier__ = 'com.chantasticvfx'
12
13    # initial default node name.
14    NODE_NAME = 'Foo Node'
15
16    def __init__(self):
17        super(FooNode, self).__init__()
18
19        # create an input port.
20        self.add_input('in', color=(180, 80, 0))
21
22        # create an output port.
23        self.add_output('out')
24
25
26if __name__ == '__main__':
27    app = QtWidgets.QApplication(sys.argv)
28
29    # create node graph controller.
30    graph = NodeGraph()
31
32    # set up default menu and commands.
33    setup_context_menu(graph)
34
35    # register the FooNode node class.
36    graph.register_node(FooNode)
37
38    # show the node graph widget.
39    graph_widget = graph.widget
40    graph_widget.show()
41
42    # create two nodes.
43    node_a = graph.create_node('com.chantasticvfx.FooNode', name='node A')
44    node_b = graph.create_node('com.chantasticvfx.FooNode', name='node B', pos=(300, 50))
45
46    # connect node_a to node_b
47    node_a.set_output(0, node_b.input(2))
48
49    app.exec_()
